  coltext=black,
  title=#1,
  fonttitle=\bfseries\sffamily\normalsize,
  coltitle=white,
  boxrule=1.5pt,
  arc=3pt,

  right=8pt,
  top=8pt,
  bottom=8pt,
  titlerule=0pt,
  toptitle=4pt,
  bottomtitle=4pt,

  boxed title style={
    colback=accentgold,
    colframe=accentgold,
    arc=2pt,
    boxrule=1pt
  }
}
